def execute_sell(slug, outcome, amount_usd, price=None):
    """
    Place a sell order (exit position early) via CLOB SDK with proper signing.
    Uses LIMIT order when price is provided (maker = zero fees + 20% rebate).
    """
    _poly_rate_limiter.wait_if_needed()

    # FIX (Apr 20): Polymarket rejects orders with size < 5 tokens. Attempting
    # stop-loss on positions that have already decayed below that floor just
    # burns API calls and logs errors. Skip cleanly and close as LOST.
    sell_price_est = price if price else 0.99
    size_est = amount_usd / sell_price_est if sell_price_est > 0 else 0
    if size_est < 5.0:
        log.info(
            "execute_sell: size {:.2f} below $5 min — skipping sell, position will auto-resolve"
            .format(size_est)
        )
        return None

    client = _get_clob_client()
    if not client:
        log.error("execute_sell: ClobClient not available (missing POLY_PRIVATE_KEY?)")
        return None

    if not _ensure_clob_reachable():
        log.error("execute_sell HALTED — proxy is down, refusing to submit order without proxy")
        return None

    token_id, neg_risk = _resolve_token_id(slug, outcome)
    if not token_id:
        return None

    order_type_rec = platform_rules.get_optimal_order_type(
        "polymarket", price or 0.99, amount_usd, _get_bankroll()
    )
    use_limit = (order_type_rec == "limit" and price is not None)

    try:
        from py_clob_client.clob_types import OrderArgs, PartialCreateOrderOptions
        from py_clob_client.order_builder.constants import SELL

        sell_price = price if price else 0.99
        size = round(amount_usd / sell_price, 2)

        try:
            tick_size = client.get_tick_size(token_id)
        except Exception:
            tick_size = "0.01"

        def _place():
            with _trade_busy():
                if _clob_throttle_mod:
                    _clob_throttle_mod.throttle(wallet_id=(MY_WALLET or "default"))
                return client.create_and_post_order(
                    OrderArgs(
                        token_id=token_id,
                        price=sell_price,
                        size=size,
                        side=SELL,
                    ),
                    options=PartialCreateOrderOptions(tick_size=tick_size, neg_risk=neg_risk or False),
                )

        try:
            resp = _place()
        except Exception as e1:
            # Structural fix (Apr 22): "not enough balance" almost always means
            # a prior sell is still pending and has the collateral locked. Cancel
            # all orders for this specific token and retry once before giving up.
            if "not enough balance" in str(e1).lower():
                try:
                    cancelled = client.cancel_market_orders(asset_id=token_id)
                    log.warning(f"execute_sell: cancelled stuck orders on {slug} ({outcome}): {cancelled}")
                    time.sleep(1.5)
                    resp = _place()
                except Exception as e2:
                    log.error(f"execute_sell retry failed after cancel: {e2}")
                    return None
            else:
                raise
        if resp and resp.get("orderID"):
            otype = "LIMIT/GTC" if use_limit else "MARKET/FOK"
            log.info(f"Sell order placed ({otype}): {resp}")
            return resp
        else:
            log.error(f"Sell response unexpected: {resp}")
            return None
    except Exception as e:
        log.error(f"execute_sell exception: {e}")
        return None

def _update_trade_status(slug, outcome, new_status, pnl=0.0):
    """Rewrite trades.jsonl updating the first matching OPEN trade to new_status.
    Also auto-reinvests the P&L into the bankroll.
    2026-04-25: added fcntl.flock on .trades.jsonl.lock so the read-modify-write
    cycle is atomic against concurrent _append_jsonl writers and auditors.
    Without it, an append between _load_jsonl and the rewrite would be silently
    lost when this function overwrites the file."""
    import fcntl as _fcntl
    nc_loss_trade = None
    anti_nc_trade = None
    anti_nc_status = None
    lock_path = str(TRADES_FILE.parent / ".trades.jsonl.lock")

    with _file_lock:
        with open(lock_path, "w") as _lk:
            _fcntl.flock(_lk, _fcntl.LOCK_EX)
            try:
                trades = _load_jsonl(TRADES_FILE)
                updated = False
                for t in trades:
                    if t.get("slug") == slug and t.get("outcome") == outcome and t.get("status") == "OPEN":
                        t["status"] = new_status
                        t["pnl"]    = round(pnl, 2)
                        t["closed_at"] = _utc_now_iso()
                        updated = True
                        break
                if updated:
                    with open(TRADES_FILE, "w") as f:
                        for t in trades:
                            f.write(json.dumps(t) + "\n")
            finally:
                _fcntl.flock(_lk, _fcntl.LOCK_UN)
        if updated:
            new_bankroll = _update_bankroll(pnl)
            global BANKROLL
            BANKROLL = new_bankroll
            log.info(f"Bankroll updated: {'+' if pnl>=0 else ''}${pnl:.2f} → ${new_bankroll:.2f} total")

            if new_status == "LOST" and pnl < 0:
                lost_t = next((t for t in trades if t.get("slug")==slug and t.get("outcome")==outcome), None)
                if lost_t and lost_t.get("type") == "NEAR_CERTAIN":
                    nc_loss_trade = lost_t

            settled_t = next((t for t in trades if t.get("slug")==slug and t.get("outcome")==outcome), None)
            if settled_t and settled_t.get("type") == "ANTI_NC":
                anti_nc_trade = settled_t
                anti_nc_status = new_status

    # Post-lock callbacks: these acquire _file_lock internally, must run OUTSIDE the lock
    if nc_loss_trade is not None:
        _analyze_nc_loss_gap(nc_loss_trade)
    if anti_nc_trade is not None:
        _update_anti_nc_wallet_stats(anti_nc_trade, anti_nc_status)

    return updated

def _normalize_title(title: str) -> str:
    """
    Normalize a market title for comparison:
    - lowercase
    - remove dates in common formats (Jan 2026, 2026-04-15, Q1 2026, etc.)
    - remove platform-specific prefixes/suffixes
    - collapse whitespace
    - strip punctuation except key symbols
    """
    if not title:
        return ""
    t = title.lower().strip()

    # Remove common date patterns
    t = re.sub(r'\b(?:jan|feb|mar|apr|may|jun|jul|aug|sep|oct|nov|dec)[a-z]*\.?\s*\d{1,2},?\s*\d{4}\b', ' ', t)
    t = re.sub(r'\b\d{4}[-/]\d{2}[-/]\d{2}\b', ' ', t)
    t = re.sub(r'\b(?:q[1-4])\s*\d{4}\b', ' ', t)
    t = re.sub(r'\bby\s+(?:jan|feb|mar|apr|may|jun|jul|aug|sep|oct|nov|dec)[a-z]*\.?\s*\d{1,4}\b', ' ', t)
    # Remove standalone years only when they look like date context (e.g., "in 2026")
    t = re.sub(r'\b(?:in|by|before|after|during)\s+20\d{2}\b', ' ', t)

    # Remove platform-specific boilerplate
    for prefix in ["will ", "will the ", "what will ", "who will "]:
        if t.startswith(prefix):
            t = t[len(prefix):]

    # Strip common suffixes
    t = re.sub(r'\?$', '', t)

    # Remove punctuation (keep alphanumeric, spaces, $, %, >, <)
    t = re.sub(r'[^\w\s$%><]', ' ', t)

    # Collapse whitespace
    t = re.sub(r'\s+', ' ', t).strip()

    return t
